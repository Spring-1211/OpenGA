\chapter{Introduction}
\label{ch:intro}

\begin{theorem}[Main Theorem]\label{thm:main}
Let $(M^{n+1}, g)$ be a complete Riemannian manifold with non-empty smooth boundary $\partial M$, $2 \leq n \leq 6$.  Let $U \subset M$ be a good set with $\overline{U} \cap \partial M \neq \varnothing$.  Then there exists a smooth, almost properly embedded free boundary minimal hypersurface $(\Sigma,\partial\Sigma)\subset(M,\partial M)$ with $\partial\Sigma\subset\partial M$ and $\Sigma$ meeting $\partial M$ orthogonally, satisfying
\[
\mathcal{H}^n(\Sigma)\le W_\partial(U)+\mathbf{M}(\partial_IU)
\qquad\text{and}\qquad
\Sigma\cap\overline{U}\neq\varnothing.
\]
\end{theorem}
