\chapter{Morse Foliations with Controlled Area of Fibers}\label{ch:morse-foliation}

The goal of this chapter is to develop technical tools for constructing \emph{nested} sweepouts from arbitrary sweepouts. The central problem is the following: given a sweepout $\{\Omega_t\}$ that may not be nested (i.e., $\Omega_s \not\subset \Omega_t$ for $s < t$), how can we deform it into a nested sweepout while controlling the area of the level sets?

A smooth function $f\colon M \to \mathbb{R}$ is \emph{$\partial$-transverse}\label{def:boundary-transverse} if $f|_{\partial M}$ has no critical points, i.e.\ $\nabla_{\partial M}(f|_{\partial M}) \neq 0$ everywhere on $\partial M$; a \emph{$\partial$-transverse Morse function} is a Morse function on $M$ that is also $\partial$-transverse. By standard transversality theory (Hirsch~\cite[Chapter~4]{Hir76}), the set of $\partial$-transverse Morse functions is residual in $C^\infty(M, \mathbb{R})$, since both the Morse condition and $\partial$-transversality are open and dense. Moreover, $\partial$-transversality is preserved under sufficiently small perturbations; since the perturbations in the proofs below are either supported in $\operatorname{int}(M)$ or arbitrarily small, the $\partial$-transverse condition is maintained throughout.

The $\partial$-transverse condition ensures that level sets $f^{-1}(t)$ meet $\partial M$ transversally, so $\mathcal{H}^n(f^{-1}(t) \cap \partial M) = 0$ and the area $\mathcal{H}^n(f^{-1}(t))$ counts only the interior part. Consequently, all results in this chapter follow Chambers--Liokumovich~\cite[\S3--4]{CL20} without essential modification. The key technical results are Lemma~\ref{lem:morse-foliation-1} and Lemma~\ref{lem:morse-foliation-non-compact}, which show that given any $\partial$-transverse Morse function $f$ and a target hypersurface $\Sigma$, one can perturb $f$ to a new Morse function $g$ having $\Sigma$ as a level set while preserving area bounds. These local constructions are glued together in Chapter~\ref{ch:nested-sweepout} to produce global nested sweepouts.

\begin{lemma}\label{lem:morse-foliation-1}
    Let $N^{n + 1} \subset M^{n + 1}$ be a compact submanifold with non-empty boundary. Let $f: N \to [a, b]$ be a $\partial$-transverse Morse function. Let $\Sigma^n$ be a closed submanifold of $\partial N$.

    For every $\eps > 0$, there exists a Morse function $g: N \to [a, b]$, such that the following holds:
    \begin{enumerate}
        \item[(1)] $g^{-1}(b) = \Sigma$;
        \item[(2)] $f^{-1}([a, t)) \subset N_{\eps/2}(g^{-1}([a, t))) \subset N_\eps(f^{-1}([a, t)))$;
        \item[(3)] $\mathcal{H}^n(g^{-1}(t)) \leq \mathcal{H}^n(\partial_I f^{-1}([a, t])) + \eps$;
        \item[(4)] If $\dist(x, f(\Sigma)) > \eps$ then $f^{-1}(x) = g^{-1}(x)$.
    \end{enumerate}
\end{lemma}
We will also need a slightly different version of this lemma for a non-compact submanifold $N$.
\begin{lemma}\label{lem:morse-foliation-non-compact}
    Let $N \subset M$ be a not necessarily compact submanifold with non-empty boundary and $f: N \to (-\infty, b]$ be a proper $\partial$-transverse Morse function. Let $\Sigma$ be a compact submanifold of $\partial N$.

    For every $\eps > 0$ there exists a Morse function $g: N \to (-\infty, b]$, such that the following holds:
    \begin{enumerate}
        \item[(1)] $g^{-1}(b) = \Sigma$;
        \item[(2)] $f^{-1}((-\infty, t)) \subset N_{\eps/2}(g^{-1}((-\infty, t))) \subset N_\eps(f^{-1}((-\infty, t)))$;
        \item[(3)] $\mathcal{H}^n(g^{-1}(t)) \leq \mathcal{H}^n(\partial_I f^{-1}((-\infty, t])) + \eps$;
        \item[(4)] If $\dist(x, f(\Sigma)) > \eps$ then $f^{-1}(x) = g^{-1}(x)$.
    \end{enumerate}
\end{lemma}
\begin{lemma}\label{lem:morse-separate}
    Let $N^{n + 1}\subset M^{n + 1}$ be a manifold with compact boundary $\partial N$ and $\Sigma$ be a hypersurface with $\partial\Sigma \subset \partial N$ and such that $\Sigma$ intersects $\partial N$ transversally and separates $N$ into two disjoint regions $N\backslash \Sigma = V_1 \sqcup V_2$.

    For every $\eps > 0$, there exist open sets with smooth boundary $\Omega_1$ and $\Omega_2$ and a Morse function $g: \operatorname{cl}(N\backslash(\Omega_1\cup\Omega_2)) \to [0, 1]$, such that the following holds:
    \begin{enumerate}
        \item[(1)] $\Omega_1$ is a relative inward $\eps$-perturbation of $V_1$; $\Omega_2$ is a relative inward $\eps$-perturbation of $V_2$.
        \item[(2)] $g^{-1}(0) = \partial\Omega_1\cup\partial\Omega_2$ and $g^{-1}(1) = \partial N$;
        \item[(3)] $\mathcal{H}^n(g^{-1}(t)) \leq \mathbf{M}(\partial_I N) + 2\,\mathcal{H}^n(\Sigma) + \eps$.
    \end{enumerate}
\end{lemma}
\begin{proof}
        Let $f: M \to \R$ be a $\partial$-transverse Morse function such that there exists some sufficiently small $0 < \delta_1 < \eps$,
        \begin{equation*}
            f(x) = \operatorname{sdist}(x, \partial N)
        \end{equation*}
        for $x \in N_{\delta_1}(\partial N)$. The sign is chosen so that $f(x) \geq 0$ when $x \in N$. Without any loss of generality, we may assume the following three conditions hold:
        \begin{enumerate}
            \item[(i)] $f$ has no critical points in the $\eps$-neighborhood of $\Sigma$: a generic Morse function has no critical points on any given compact set, so a small perturbation of $f$ (supported away from $N_{\delta_1}(\partial N)$) achieves this.
            \item[(ii)] The restriction $f|_\Sigma$ is also Morse: by Sard's theorem, $f|_\Sigma$ is Morse for a generic choice of $f$.
            \item[(iii)] The mass bound
        \begin{equation*}
            \mathcal{H}^n(f^{-1}(t)) \leq \mathbf{M}(\partial_I N) + \eps/10
        \end{equation*}
        holds for $t \in [0, \delta_1]$: since $f = \operatorname{sdist}(\cdot, \partial N)$ in $N_{\delta_1}(\partial N)$, the level set $f^{-1}(t)$ converges to $\partial N$ in the $C^1$ sense as $t \to 0$, so $\mathcal{H}^n(f^{-1}(t) \cap N)$ converges to $\mathcal{H}^n(\partial_I N)$, and we can choose $\delta_1$ small enough for this bound to hold.
        \end{enumerate} Pick $\delta_2 \in (0, \delta_1)$ so that for $t \in (0, \delta_2)$,
        \begin{equation*}
            \mathcal{H}^n(f^{-1}(t)\cap N_{\delta_2}(\Sigma)) < \eps/10
        \end{equation*}
        and
        \begin{equation*}
            \mathcal{H}^n(\partial N_{\delta_2}(\Sigma)) \leq 2\,\mathcal{H}^n(\Sigma) + \eps/10.
        \end{equation*}

        Let $\Omega_i$ denote a relative inward $\delta_2/2$-perturbation of $V_i$, i.e.
        \begin{enumerate}
            \item[(a)] $V_i\backslash N_{\delta_2/2}(\partial V_i) \subset \Omega_i \subsetneq V_i$;
            \item[(b)] there exists a nested family of open sets $\br{\Xi_t^i}_{t \in [0, 1]}$ and corresponding smooth isotopies $\partial\Xi_t^i$, such that $\partial\Xi_1^i$ is a smoothing of $\partial V_i$ and
            \begin{equation*}
                \mathcal{H}^n(\partial\Xi_t^i) \leq \mathcal{H}^n(\partial V_i) + \delta_2/2
            \end{equation*}
            for all $t \in [0, 1]$.
        \end{enumerate}

        We can make an arbitrarily small perturbation of $f$ by an ambient diffeomorphism so that $f$ is a Morse function on $\partial\Omega_1\cup\partial\Omega_2$. For $t \in [0, \delta_2]$, since $f = \operatorname{sdist}(\cdot, \partial N)$ in $N_{\delta_1}(\partial N)$, the level set $f^{-1}(t)$ is $C^1$-close to $\partial N$. In particular, each leaf $\Sigma_t = (f^{-1}(t)\cap N)\backslash(\Omega_1\cup\Omega_2)$ is a normal graph over some region $U_t\subset \partial N$, with

        \begin{equation*}
            \mathcal{H}^n(\Sigma_t) \leq \mathcal{H}^n(U_t) + \eps/5
        \end{equation*}

        It follows that
        \begin{equation*}
            \mathcal{H}^n(f^{-1}(t) \cap \operatorname{cl}(N\setminus(\Omega_1\cup\Omega_2))) \leq 2\,\mathcal{H}^n(\Sigma) + \mathbf{M}(\partial_I N) + \eps/2.
        \end{equation*}
        We now verify that the hypotheses of Lemma~\ref{lem:morse-foliation-non-compact} are satisfied for the restriction $f|_{N\backslash(\Omega_1\cup\Omega_2)}$:
        \begin{itemize}
            \item The domain $N' = \operatorname{cl}(N \setminus (\Omega_1 \cup \Omega_2))$ is a (non-compact) manifold with boundary $\partial N' = \partial\Omega_1 \cup \partial\Omega_2 \cup \partial N$.
            \item $f|_{N'}$ is proper: preimages of compact sets in $[0, \delta_1]$ are compact since the non-compact ends (i.e.\ the interiors of $V_1$ and $V_2$) have been removed by excising $\Omega_1$ and $\Omega_2$.
            \item $f|_{N'}$ is $\partial$-transverse on $N'$, as this is inherited from the $\partial$-transversality of $f$ on $N$.
            \item $\partial\Omega_1 \cup \partial\Omega_2$ plays the role of $\Sigma$ in Lemma~\ref{lem:morse-foliation-non-compact}, i.e.\ the compact submanifold of $\partial N'$ sent to $b = \delta_1$.
        \end{itemize}
        Applying Lemma~\ref{lem:morse-foliation-non-compact}, we obtain the desired function $g$ and the corresponding foliation, i.e.
        \begin{equation*}
            \mathcal{H}^n(g^{-1}(t)) \leq 2\,\mathcal{H}^n(\Sigma) + \mathbf{M}(\partial_I N) + \eps.
        \end{equation*}
    \end{proof}
