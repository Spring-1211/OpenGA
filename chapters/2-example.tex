\chapter{Settings and Preliminaries}\label{ch:preliminaries}

This chapter establishes the foundational framework for the free boundary min-max theory in complete Riemannian manifolds with boundary.

Let $(M^{n+1}, g)$ be a smooth, complete Riemannian manifold with non-empty smooth boundary $\partial M$. We assume $2 \leq n \leq 6$ throughout. We write $\nu_{\partial M}$ for the inward unit normal to $\partial M$. See Appendix~\ref{sec:fermi-coordinates} for details on Fermi coordinates near $\partial M$.

We denote by $\mathcal{H}^k$ the $k$-dimensional Hausdorff measure and by $\mathbf{M}(T)$ the mass of a current $T$. We assume $M$ is isometrically embedded in some $\mathbb{R}^L$. For vector fields, let $\mathfrak{X}_c(M)$ denote the space of \emph{compactly supported} smooth vector fields on $M$ that are tangent to $M$:
\begin{multline*}
\mathfrak{X}_c(M) := \bigl\{X \in \mathfrak{X}(\mathbb{R}^L) : X \text{ is compactly supported},\\
X(p) \in T_p M \text{ for all } p \in M\bigr\}.
\end{multline*}
The subspace of vector fields additionally tangent to the boundary is
\[
\mathfrak{X}_{\mathrm{tan}}(M) := \{X \in \mathfrak{X}_c(M) : X(p) \in T_p(\partial M) \text{ for all } p \in \partial M\}.
\]

For a relatively open subset $U \subset M$, we write $\mathrm{int}_M(U)$ for the relative interior of $U$ in $M$. We use $A \Subset B$ to denote that $\overline{A}$ is compact and contained in $B$.

\begin{definition}[$\partial$-transverse function]\label{def:boundary-transverse}
A smooth function $f: M \to \mathbb{R}$ is \emph{$\partial$-transverse} if $f|_{\partial M}$ has no critical points, i.e.\ $\nabla_{\partial M}(f|_{\partial M}) \neq 0$ everywhere on $\partial M$.  A \emph{$\partial$-transverse Morse function} is a Morse function on $M$ that is also $\partial$-transverse.
\end{definition}

\section{Currents and Varifolds}\label{sec:currents-varifolds}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

We work with currents and relative cycles following Li--Zhou~\cite{LZ17}. We write $\mathcal{R}_k(M;\mathbb{Z}_2)$ for the space of mod~$2$ flat chains supported in $M$, with mass norm $\mathbf{M}$ and flat topology. For a bounded open set $\Omega \subset M$ of finite perimeter (i.e.\ a Caccioppoli set), we write $Q_\Omega := \llbracket \Omega \rrbracket \in \mathcal{R}_{n+1}(M;\mathbb{Z}_2)$ for the associated $(n+1)$-current, so that $\mathbf{M}(\partial Q_\Omega) < \infty$.

\begin{remark}[Coefficient group]\label{rmk:Z2}
Throughout this thesis we work with $\mathbb{Z}_2$ coefficients. We write $\mathcal{R}_k(M;\mathbb{Z}_2)$ for the space of mod~$2$ flat chains supported in $M$, and all cycle spaces $\mathcal{Z}_n(A,B;\mathbb{Z}_2)$, $\mathbf{Z}_n(A,B;\mathbb{Z}_2)$ are defined with $\mathbb{Z}_2$ coefficients. Li--Zhou~\cite{LZ17} originally state their results for $\mathbb{Z}$ coefficients (integer currents), but the key tools---the isoperimetric lemma, slicing theorem, and regularity theory---hold equally for $\mathbb{Z}_2$ coefficients (see \cite{MN17,Alm62}). The choice of $\mathbb{Z}_2$ has the advantage that every Caccioppoli set $\Omega \subset M$ of finite perimeter naturally induces a mod~$2$ cycle $\llbracket \partial\Omega \rrbracket \in \mathcal{Z}_n(M,\partial M;\mathbb{Z}_2)$ with
\[
  \mathrm{Per}(\Omega) = \mathbf{M}(\llbracket \partial\Omega \rrbracket),
\]
since the reduced boundary $\partial^*\Omega$ carries no orientation, only a distinction between interior and exterior. We also note that the regularity theory (Theorem~\ref{thm:main-regularity}) operates at the varifold level and does not depend on the coefficient group: replacement varifolds arising from mod~$2$ area minimization are integer rectifiable and stable, so Wickramasekera's theory~\cite{Wic14} applies without modification.
\end{remark}

\begin{proposition}[Caccioppoli sets and mod~$2$ cycles]\label{prop:cacc-cycle}
Let $\Omega \subset M$ be a Caccioppoli set with $\mathrm{Per}(\Omega) < \infty$. Then the reduced boundary $\partial^*\Omega$ induces a mod~$2$ relative cycle
\[
  \llbracket \partial\Omega \rrbracket \in \mathcal{Z}_n(M, \partial M;\mathbb{Z}_2)
\]
satisfying $\mathrm{Per}(\Omega) = \mathbf{M}(\llbracket\partial\Omega\rrbracket)$. Moreover, the varifold $|\partial^*\Omega|$ induced by the reduced boundary agrees with the varifold $|\llbracket\partial\Omega\rrbracket|$ induced by the mod~$2$ current.
\end{proposition}

\begin{proof}
This follows from the structure theorem for Caccioppoli sets; see Maggi~\cite[Theorem~15.9]{Mag12}. The reduced boundary $\partial^*\Omega$ is countably $\mathcal{H}^n$-rectifiable, and the associated multiplicity-one current $\llbracket\partial\Omega\rrbracket$ satisfies $\partial\llbracket\partial\Omega\rrbracket = 0$ in $\mathrm{int}(M)$ (in the mod~$2$ sense). Since $\partial^*\Omega$ carries no canonical orientation, the natural coefficient group is $\mathbb{Z}_2$. \textcolor{red}{The mass identity $\mathrm{Per}(\Omega) = \mathbf{M}(\llbracket\partial\Omega\rrbracket)$ is immediate from the definitions.}\xinze{Need more justification, this seems trivial but haven't been done anywhere else}
\end{proof}

\begin{definition}[Relative cycles and equivalence classes]\label{def:relative-cycle}
For subsets $A \supset B$ of $M$, the space of \emph{relative cycles} is
\[
\mathcal{Z}_n(A, B;\mathbb{Z}_2) := \{T \in \mathcal{R}_n(M;\mathbb{Z}_2) : \mathrm{spt}(T) \subset A, \ \mathrm{spt}(\partial T) \subset B\},
\]
equipped with the flat topology. The \emph{space of equivalence classes of relative cycles} is the quotient
\[
\mathbf{Z}_n(A, B;\mathbb{Z}_2) := \mathcal{Z}_n(A, B;\mathbb{Z}_2) / \mathcal{Z}_n(B, B;\mathbb{Z}_2).
\]
Two cycles $T_1, T_2 \in \mathcal{Z}_n(A, B;\mathbb{Z}_2)$ represent the same class $\tau \in \mathbf{Z}_n(A, B;\mathbb{Z}_2)$ if and only if $T_1 - T_2 \in \mathcal{Z}_n(B, B;\mathbb{Z}_2)$, i.e., $T_1 - T_2$ is supported in $B$.
\end{definition}

\begin{definition}[Canonical representative]\label{def:canonical-rep}
For $\tau \in \mathbf{Z}_n(A, B;\mathbb{Z}_2)$, there exists a unique representative $T \in \tau$ satisfying $T \llcorner B = 0$. This $T$ is called the \emph{canonical representative} of $\tau$. When no confusion arises, we identify $\tau$ with its canonical representative.
\end{definition}

\begin{definition}[Mass of an equivalence class]\label{def:mass-equivalence-class}
For $\tau \in \mathbf{Z}_n(A, B;\mathbb{Z}_2)$, the \emph{mass} of $\tau$ is defined as $\mathbf{M}(\tau) := \mathbf{M}(T)$, where $T$ is the canonical representative of $\tau$.
\end{definition}

\begin{remark}\label{rem:mass-infimum}
By \cite[Lemma~3.3]{LZ17}, we have $\mathbf{M}(\tau) = \inf_{S \in \tau} \mathbf{M}(S)$. In particular, the canonical representative achieves the infimum of mass over all representatives.
\end{remark}

\begin{definition}[Relative flat norm]\label{def:relative-flat-norm}
For a compact set $K \subset M$ and $\tau \in \mathbf{Z}_n(A, B;\mathbb{Z}_2)$, the \emph{relative flat norm} is
\[
\mathcal{F}^K(\tau) := \inf_{T \in \tau} \mathcal{F}^K(T),
\]
where $\mathcal{F}^K(T) := \inf\{\mathbf{M}(T - \partial S) + \mathbf{M}(S) : S \in \mathcal{R}_{n+1}(M;\mathbb{Z}_2), \mathrm{spt}(S) \subset K\}$ is the flat norm of $T$ relative to $K$.
\end{definition}

\begin{lemma}[$\mathcal{F}$-isoperimetric lemma {\cite[Lemma~3.15]{LZ17}}]\label{lem:F-isoperimetric}
There exist $\epsilon_M > 0$ and $C_M \geq 1$ depending only on the isometric embedding $M \hookrightarrow \mathbb{R}^L$ such that for any $\tau_1, \tau_2 \in \mathbf{Z}_n(M, \partial M;\mathbb{Z}_2)$ with $\mathcal{F}(\tau_2 - \tau_1) < \epsilon_M$, there exists a mod~$2$ $(n+1)$-current $Q$ (an \emph{isoperimetric choice}) such that:
\begin{itemize}
    \item $\mathrm{spt}(Q) \subset M$;
    \item $\mathrm{spt}(T_2 - T_1 - \partial Q) \subset \partial M$;
    \item $\mathbf{M}(Q) \leq C_M \mathcal{F}(\tau_2 - \tau_1)$,
\end{itemize}
where $T_1, T_2$ are the canonical representatives of $\tau_1, \tau_2$.
\end{lemma}

\begin{lemma}[$\mathbf{M}$-isoperimetric lemma {\cite[Lemma~3.17]{LZ17}}]\label{lem:M-isoperimetric}
There exist $\epsilon_M > 0$ and $C_M \geq 1$ depending only on the isometric embedding $M \hookrightarrow \mathbb{R}^L$ such that for any $\tau_1, \tau_2 \in \mathbf{Z}_n(M, \partial M;\mathbb{Z}_2)$ with $\mathbf{M}(\tau_2 - \tau_1) < \epsilon_M$, there exist a mod~$2$ $(n+1)$-current $Q$ and a mod~$2$ rectifiable $n$-current $R$ such that:
\begin{itemize}
    \item $\mathrm{spt}(Q) \subset M$, $\mathrm{spt}(R) \subset \partial M$;
    \item $T_2 - T_1 = \partial Q + R$;
    \item $\mathbf{M}(Q) + \mathbf{M}(R) \leq C_M \mathbf{M}(\tau_2 - \tau_1)$,
\end{itemize}
where $T_1, T_2$ are the canonical representatives.
\end{lemma}

\begin{remark}[Isoperimetric constants in the non-compact setting]\label{rem:isoperimetric-local}
In Lemmas~\ref{lem:F-isoperimetric} and~\ref{lem:M-isoperimetric}, the constants $\epsilon_M$ and $C_M$ depend on the isometric embedding $M\hookrightarrow\mathbb{R}^L$ through Almgren's isoperimetric inequality for compact Lipschitz neighbourhood retracts (CLNRs)~\cite{Alm62}. When $M$ is complete but non-compact, these lemmas are applied only to cycles whose supports are contained in a precompact region $K\subset M$. Since $\overline{K}$ is compact, it admits an isometric embedding $\overline{K}\hookrightarrow\mathbb{R}^L$ as a CLNR, and the constants $\epsilon_{\overline{K}},C_{\overline{K}}$ are well-defined and positive. In the exhaustion argument of Chapter~\ref{ch:convergence-of-min-max}, each compact set $\overline{U}_j$ has its own isoperimetric constants; the diagonal argument only requires these constants to be positive for each fixed $j$, not uniformly bounded over the exhaustion.
\end{remark}

\begin{remark}\label{rem:filling-cycle}
For any bounded open set $\Omega \subset M$ of finite perimeter, the current $\partial \llbracket \Omega \rrbracket$ gives a relative cycle in $\mathcal{Z}_n(M, \partial M;\mathbb{Z}_2)$. We refer to $\llbracket \Omega \rrbracket$ as the \emph{filling} of this cycle. For a relative cycle $\tau \in \mathbf{Z}_n(M, \partial M;\mathbb{Z}_2)$ with canonical representative $T$, the mass restricted to an open set $V$ is $\|\tau\|(V) := \mathbf{M}(T \llcorner V)$.
\end{remark}

For a bounded open set $U \subset M$ with smooth boundary, the associated $(n+1)$-current $\llbracket U \rrbracket \in \mathcal{R}_{n+1}(M;\mathbb{Z}_2)$ has boundary
\[
\partial \llbracket U \rrbracket = \partial_I U + \partial_F U,
\]
where:
\begin{itemize}
    \item $\partial_I U := \partial \llbracket U \rrbracket \llcorner \mathrm{int}(M)$ is the \emph{interior boundary}, supported on $\partial U \cap \mathrm{int}(M)$;
    \item $\partial_F U := \partial \llbracket U \rrbracket \llcorner \partial M$ is the \emph{free boundary part}, supported on $\overline{U} \cap \partial M$.
\end{itemize}

\begin{remark}\label{rem:LZ-correspondence}
In the notation of Li--Zhou~\cite{LZ17}, sweepouts of $(U, \partial_F U)$ correspond to families in $\mathbf{Z}_n(M, \partial M;\mathbb{Z}_2)$ (Li--Zhou originally use $\mathbf{Z}_n(\overline{U}, \overline{U} \cap \partial M)$ with $\mathbb{Z}$ coefficients; see Remark~\ref{rmk:Z2}). The interior boundary $\partial_I U$ is a relative cycle in $\mathcal{Z}_n(M, \partial M;\mathbb{Z}_2)$ since $U$ has smooth boundary.
\end{remark}

\begin{remark}[Why equivalence classes are necessary]\label{rem:why-equivalence}
Working with the quotient space $\mathbf{Z}_n(A, B;\mathbb{Z}_2)$ rather than the space of mod~$2$ flat chains $\mathcal{Z}_n(A, B;\mathbb{Z}_2)$ is essential for several reasons:
\begin{enumerate}[label=\textup{(\roman*)}]
    \item \emph{Well-defined mass}: The mass $\mathbf{M}(\tau)$ of an equivalence class (Definition~\ref{def:mass-equivalence-class}) is canonically defined via the canonical representative, whereas the mass of a current in a class depends on the choice of representative.
    \item \emph{Flat continuity}: In the proof of Proposition~\ref{prop:sweepout-cycle}, the boundary residue $R$ supported on $\partial M$ vanishes in the equivalence class, which is essential for the $\mathcal{F}$-continuity of the map $t \mapsto \llbracket\partial\Omega_t\rrbracket$.
    \item \emph{Isoperimetric lemmas}: The $\mathcal{F}$- and $\mathbf{M}$-isoperimetric lemmas (Lemmas~\ref{lem:F-isoperimetric} and~\ref{lem:M-isoperimetric}) require working with equivalence classes to ensure that the isoperimetric choices are well-defined up to boundary terms.
\end{enumerate}
\end{remark}

We recall the slicing theory for mod~$2$ flat chains; see Simon~\cite[Sections~28--29]{Sim84}.

\begin{theorem}[Slicing~{\cite[28.4, 28.5]{Sim84}}]\label{thm:slicing}
Let $T \in \mathcal{R}_{n+1}(M;\mathbb{Z}_2)$ and let $f: M \to \mathbb{R}$ be a Lipschitz function. Then for almost every $t \in \mathbb{R}$, the \emph{slice}
\[
\langle T, f, t \rangle \in \mathcal{R}_n(M;\mathbb{Z}_2)
\]
is a well-defined mod~$2$ rectifiable $n$-current satisfying:
\begin{enumerate}[label=\textup{(\roman*)}]
    \item (Boundary formula) $\partial \langle T, f, t \rangle = \langle \partial T, f, t \rangle$ wherever $\partial T$ has locally finite mass;
    \item (Coarea inequality) $\int_{\mathbb{R}} \mathbf{M}(\langle T, f, t \rangle)\, dt \leq \int |\nabla f|\, d\|T\|$.
\end{enumerate}
\end{theorem}

\begin{corollary}[Coarea formula for fillings]\label{cor:coarea-filling}
Let $Q = \llbracket \Omega \rrbracket$ for a bounded open set $\Omega \subset M$ of finite perimeter, and let $f: M \to \mathbb{R}$ be Lipschitz. Then for almost every $t$, the slice $\langle Q, f, t \rangle \in \mathcal{Z}_n(M, \partial M;\mathbb{Z}_2)$ is a relative cycle, and
\[
\int_{\mathbb{R}} \mathbf{M}(\langle Q, f, t \rangle)\, dt \leq \int_\Omega |\nabla f|\, d\mathcal{H}^{n+1}.
\]
\end{corollary}

We collect the basic definitions from varifold theory needed in later chapters. Standard references include Pitts~\cite{Pit81} and Simon~\cite{Sim84}.

\begin{definition}[Varifolds]\label{def:varifold}
Let $\mathrm{Gr}_n(M)$ denote the Grassmannian bundle of (unoriented) $n$-planes over $M$. An \emph{$n$-varifold} on $M$ is a Radon measure on $\mathrm{Gr}_n(M)$. The space of all $n$-varifolds is denoted $\mathcal{V}_n(M)$, equipped with the weak-$*$ topology.

For $V \in \mathcal{V}_n(M)$, the \emph{mass measure} $\|V\|$ is the pushforward of $V$ under the projection $\mathrm{Gr}_n(M) \to M$. The \emph{total mass} is $\|V\|(M)$, and the \emph{support} is $\operatorname{spt}\|V\|$. For an open set $A \subset M$, the \emph{restriction} $V \llcorner A$ is the varifold defined by $V \llcorner A(E) := V(E \cap \mathrm{Gr}_n(A))$.
\end{definition}

\begin{definition}[First variation]\label{def:first-variation}
The \emph{first variation} of $V \in \mathcal{V}_n(M)$ is the linear functional on smooth vector fields defined by
\[
\delta V(X) := \int_{\mathrm{Gr}_n(M)} \operatorname{div}_S X(x) \, dV(x, S),
\]
where $\operatorname{div}_S X(x) := \sum_{i=1}^n \langle \nabla_{e_i} X, e_i \rangle$ for any orthonormal basis $\{e_i\}$ of $S$.
\end{definition}

\begin{definition}[Stationarity]\label{def:stationary}
A varifold $V \in \mathcal{V}_n(M)$ is:
\begin{enumerate}[label=\textup{(\roman*)}]
    \item \emph{stationary in $\operatorname{int}(M)$} if $\delta V(X) = 0$ for all $X \in \mathfrak{X}_c(\operatorname{int}(M))$;
    \item \emph{stationary with free boundary} if $\delta V(X) = 0$ for all $X \in \mathfrak{X}_{\mathrm{tan}}(M)$.
\end{enumerate}
\end{definition}

\begin{remark}\label{rem:stationarity-relation}
Condition~(ii) implies condition~(i) since $\mathfrak{X}_c(\operatorname{int}(M)) \subset \mathfrak{X}_{\mathrm{tan}}(M)$. The additional constraint in~(ii) is that the first variation vanishes even for vector fields that are supported up to $\partial M$, as long as they are tangent to $\partial M$.
\end{remark}

\begin{definition}[$\mathbf{F}$-metric]\label{def:F-metric}
The \emph{$\mathbf{F}$-metric} on $\mathcal{V}_n(M)$ is defined by
\[
\mathbf{F}(V, W) := \sup\biggl\{\int f \, dV - \int f \, dW : f \in C_c(\mathrm{Gr}_n(M)),\ |f| \leq 1,\ \operatorname{Lip}(f) \leq 1\biggr\}.
\]
On sets of uniformly bounded mass, $\mathbf{F}$ metrizes the weak-$*$ topology. For an open set $U \subset M$, the \emph{localized $\mathbf{F}$-metric} is $\mathbf{F}_U(V, W) := \mathbf{F}(V \llcorner U, W \llcorner U)$. See \cite[Section~2]{LZ17}.
\end{definition}

\begin{definition}[Pushforward of varifolds]\label{def:varifold-pushforward}
Let $\varphi: M \to M$ be a diffeomorphism. The \emph{pushforward} $\varphi_\sharp V \in \mathcal{V}_n(M)$ is defined by
\[
\varphi_\sharp V(A) := \int_{\mathrm{Gr}_n(M)} \mathbf{1}_A(\varphi(x), d\varphi_x(S)) \cdot |J_S \varphi(x)| \, dV(x, S)
\]
for Borel $A \subset \mathrm{Gr}_n(M)$, where $J_S \varphi(x) := \det{d\varphi_x|_S}$ is the $n$-dimensional Jacobian of $\varphi$ restricted to $S$.
\end{definition}

\section{Sweepouts and Width}\label{sec:sweepouts}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

We adopt a continuous sweepout framework based on Caccioppoli sets (sets of finite perimeter), following the approach of Chambers--Liokumovich~\cite{CL20} and De~Lellis--Tasnady~\cite{DLT13}. For applications to non-compact manifolds, we consider sweepouts of bounded open sets.

\begin{definition}[Sweepout of $(U, \partial_F U)$]\label{def:sweepout}
Let $U \subset M$ be a bounded relatively open set of finite perimeter. A \emph{sweepout of $(U, \partial_F U)$} is a family of Caccioppoli sets $\{\Omega_t\}_{t \in [0,1]}$ in $M$ satisfying:
\begin{enumerate}[label=\textup{(sw\arabic*)}]
    \item\label{sw:boundary} \textbf{Boundary condition}: $\Omega_0 \cap U = \emptyset$ and $U \subset \Omega_1$;
    \item\label{sw:continuity} \textbf{Continuity}: the map $t \mapsto \mathbf{1}_{\Omega_t}$ is continuous in $L^1(M)$, i.e.,
    \[
    \mathcal{H}^{n+1}(\Omega_t \triangle \Omega_s) \to 0 \quad \text{as } s \to t;
    \]
    \item\label{sw:finite-mass} \textbf{Finite mass}: $\sup_{t \in [0,1]} \mathrm{Per}(\Omega_t) < \infty$;
    \item\label{sw:no-concentration} \textbf{No mass concentration}: for every $x \in M$,
    \[
    \lim_{r \to 0} \sup_{t \in [0,1]} \mathrm{Per}(\Omega_t, B_r(x)) = 0.
    \]
\end{enumerate}
The \emph{height} of the sweepout is $F(\{\Omega_t\}) := \sup_{t \in [0,1]} \mathrm{Per}(\Omega_t)$.
\end{definition}

\begin{definition}[Width]\label{def:width}
The \emph{width} of $(U, \partial_F U)$ is
\[
W(U) := \inf\bigl\{ F(\{\Omega_t\}) : \{\Omega_t\} \text{ is a sweepout of } (U, \partial_F U) \bigr\}.
\]
\end{definition}

\begin{remark}\label{rem:free-boundary-automatic}
The free boundary condition is built into the Caccioppoli framework: by Proposition~\ref{prop:cacc-cycle}, each reduced boundary $\partial^*\Omega_t$ induces a mod~$2$ relative cycle $\llbracket\partial\Omega_t\rrbracket \in \mathcal{Z}_n(M, \partial M;\mathbb{Z}_2)$, whose ``boundary'' (in the sense of currents) is automatically supported on $\partial M$. Note that the sweepout ``fills'' $U$ in the sense that the fillings $\Omega_t$ transition from being disjoint from $U$ to containing $U$.
\end{remark}

The following proposition provides the bridge between the Caccioppoli sweepout framework and the cycle-valued maps used in Li--Zhou~\cite{LZ17}.

\begin{proposition}[From Caccioppoli sweepouts to cycle-valued maps]\label{prop:sweepout-cycle}
Let $\{\Omega_t\}_{t \in [0,1]}$ be a sweepout of $(U, \partial_F U)$. Then the map
\[
\Phi: [0,1] \to \mathcal{Z}_n(M, \partial M;\mathbb{Z}_2), \quad t \mapsto \llbracket\partial\Omega_t\rrbracket
\]
is continuous in the flat topology $\mathcal{F}$, and satisfies $\mathbf{M}(\Phi(t)) = \mathrm{Per}(\Omega_t)$ for all $t$ by Proposition~\ref{prop:cacc-cycle}. Moreover, $\Phi$ satisfies the no mass concentration condition
\[
\lim_{r \to 0} \sup_{t \in [0,1]} \|\Phi(t)\|(B_r(x)) = 0 \quad \text{for every } x \in M.
\]
In particular:
\begin{enumerate}[label=\textup{(\roman*)}]
    \item the width satisfies $W(U) = \inf_\Phi \sup_t \mathbf{M}(\Phi(t))$, where the infimum is over all $\mathcal{F}$-continuous maps arising from sweepouts;
    \item the results of Li--Zhou~\cite{LZ17} on pull-tight, almost minimizing selection, and the regularity theory, which are formulated at the level of continuous cycle-valued maps and varifolds, apply to $\Phi$ after discretization via~\cite[Theorem~5.1]{LZ17}.
\end{enumerate}
\end{proposition}

\begin{proof}
The mass identity $\mathbf{M}(\Phi(t)) = \mathrm{Per}(\Omega_t)$ is Proposition~\ref{prop:cacc-cycle}. The no mass concentration condition for $\Phi$ is immediate from~\ref{sw:no-concentration} and the mass identity.

For the $\mathcal{F}$-continuity: condition~\ref{sw:continuity} gives $\mathcal{F}(\llbracket\Omega_t\rrbracket - \llbracket\Omega_s\rrbracket) \leq \mathcal{H}^{n+1}(\Omega_t \triangle \Omega_s) \to 0$ as $s \to t$, since the $(n+1)$-current $\llbracket\Omega_t\rrbracket - \llbracket\Omega_s\rrbracket = \llbracket\Omega_t \setminus \Omega_s\rrbracket - \llbracket\Omega_s \setminus \Omega_t\rrbracket$ (mod~$2$) has mass equal to $\mathcal{H}^{n+1}(\Omega_t \triangle \Omega_s)$. Since $\partial(\llbracket\Omega_t\rrbracket - \llbracket\Omega_s\rrbracket) = \llbracket\partial\Omega_t\rrbracket - \llbracket\partial\Omega_s\rrbracket + R$, where $\mathrm{spt}(R) \subset \partial M$, the difference $\Phi(t) - \Phi(s)$ as a class in $\mathbf{Z}_n(M, \partial M;\mathbb{Z}_2)$ satisfies
\[
\mathcal{F}\bigl(\Phi(t) - \Phi(s)\bigr) \leq \mathcal{H}^{n+1}(\Omega_t \triangle \Omega_s) \to 0.
\]
Here we have used that $\llbracket\Omega_t\rrbracket - \llbracket\Omega_s\rrbracket$ serves as the ``filling'' in the definition of the flat norm (Definition~\ref{def:relative-flat-norm}), and the boundary residue $R$ on $\partial M$ vanishes in the equivalence class.

Part~(i) follows from the mass identity. Part~(ii) follows because $\Phi$ is $\mathcal{F}$-continuous with no mass concentration, so it satisfies the hypotheses of the Discretization Theorem~\cite[Theorem~5.1]{LZ17}, which produces a $(1, \mathbf{M})$-homotopy sequence. The pull-tight procedure~\cite[Proposition~4.17]{LZ17} then applies to this sequence, producing stationary varifolds; the almost minimizing property~\cite[Theorem~5.3]{LZ17} selects varifolds that are almost minimizing in small annuli; and the regularity theory~\cite[Theorem~6.2]{LZ17} operates entirely at the varifold level.
\end{proof}

\begin{definition}[Good sweepout]\label{def:good-sweepout}
A sweepout $\{\Omega_t\}$ of $(U, \partial_F U)$ is a \emph{good sweepout} if there exists $\varepsilon > 0$ such that
\[
\mathrm{Per}(\Omega_t) \leq 5\,\mathbf{M}(\partial_I U) \quad \text{for all } t \in [0, \varepsilon) \cup (1-\varepsilon, 1].
\]
We write $W_g(U)$ for the infimum of heights over all good sweepouts.
\end{definition}

\begin{definition}[Nested sweepout]\label{def:nested-sweepout}
A sweepout $\{\Omega_t\}$ of $(U, \partial_F U)$ is \emph{nested} if $\Omega_s \subset \Omega_t$ for all $0 \leq s \leq t \leq 1$. Equivalently, $\{\Omega_t\}$ arises from a $\partial$-transverse Morse function $f: M \to \mathbb{R}$ via $\Omega_t = f^{-1}((-\infty, c_t))$ for some increasing function $t \mapsto c_t$. We denote the collection of nested sweepouts of $U$ by $\mathcal{S}_n(U)$, and write $W_n(U) := \inf_{S \in \mathcal{S}_n(U)} F(S)$.
\end{definition}

\begin{definition}[Relative sweepout and relative width]\label{def:relative-sweepout}
A \emph{relative sweepout} of $U$ is a sweepout $\{\Omega_t\}$ of $(U, \partial_F U)$ with the additional property that $\partial^*\Omega_t \subset \overline{U}$ for all $t \in [0,1]$. We denote the collection of relative sweepouts by $\mathcal{S}_\partial(U)$, and define the \emph{relative width} as $W_\partial(U) := \inf_{S \in \mathcal{S}_\partial(U)} F(S)$.
\end{definition}

\begin{proposition}[Width inequalities]\label{prop:width-inequalities}
Let $U \subset M$ be a bounded open set with smooth boundary. Then:
\begin{enumerate}[label=\textup{(\roman*)}]
    \item $W_\partial(U) \leq W_n(U) \leq W_\partial(U) + \mathbf{M}(\partial_I U)$;
    \item $W_n(U) = W(U)$, where $W(U)$ is the width over all sweepouts;
    \item If $U$ is a good set (Definition~\ref{def:good-set}), then $W_g(U) = W(U)$.
\end{enumerate}
\end{proposition}

\begin{proof}
Part~(i) follows by restricting a nested sweepout to $\overline{U}$ (giving the lower bound) and extending a relative sweepout by adding $\partial_I U$ at each time (giving the upper bound). Part~(ii) is proved by approximating an arbitrary sweepout by nested ones using the coarea formula (Theorem~\ref{thm:slicing}); see \cite[Lemma~4.2]{CL20}. Part~(iii) follows from~(ii) by a cut-off argument using the good set condition.
\end{proof}

\begin{remark}[Equivalence with the discrete framework]\label{rmk:discrete-equiv}
Li--Zhou~\cite{LZ17} define sweepouts using the Almgren--Pitts discrete framework: an $(m, \mathbf{M})$-homotopy sequence $S = \{\phi_i\}_{i \in \mathbb{N}}$, where $\phi_i: I(m, j_i)_0 \to \mathbf{Z}_n(M, \partial M;\mathbb{Z}_2)$ are maps on the vertices of a cubical cell complex $I(m, j_i)$ satisfying fineness, boundary, and homotopy conditions (see~\cite[Definitions~5.1--5.4]{LZ17}). Their Discretization Theorem~\cite[Theorem~5.1]{LZ17} and Interpolation Theorem~\cite[Theorem~5.2]{LZ17} establish a bijection between homotopy classes of $(1, \mathbf{M})$-homotopy sequences and homotopy classes of $\mathcal{F}$-continuous maps $[0,1] \to \mathcal{Z}_n(M, \partial M;\mathbb{Z}_2)$, preserving the width. By Proposition~\ref{prop:sweepout-cycle}, our continuous Caccioppoli sweepouts give rise to such $\mathcal{F}$-continuous maps. In particular, Definitions~\ref{def:sweepout}--\ref{def:width} and the discrete formulation of~\cite{LZ17} yield the same width $W(U)$ and the same collection of min-max varifolds.
\end{remark}

\section{Good Sets}\label{sec:good-sets}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\begin{definition}[Good set]\label{def:good-set}
A bounded open subset $U \subset M$ of finite perimeter is a \emph{good set} if
\[
\mathbf{M}(\partial_I U) \leq \frac{1}{10}\, W_\partial(U),
\]
where $\partial_I U = \partial \llbracket U \rrbracket \llcorner \mathrm{int}(M)$ is the interior boundary of $U$.
\end{definition}

\begin{remark}[The bottleneck principle]\label{rem:bottleneck}
The good set condition ensures that mass cannot ``escape to infinity'' through $\partial_I U$. Any sweepout of $U$ must move the filling from $Q_0 \llcorner U = 0$ to $Q_1 \llcorner U = \llbracket U \rrbracket$. This transfer must pass through the ``bottleneck'' $\partial_I U$, whose mass satisfies $\mathbf{M}(\partial_I U) \ll W_\partial(U)$. Therefore at least one slice $\tau_t$ must carry substantial mass inside $\overline{U}$.

This principle is made rigorous in Chapter~\ref{ch:no-escape-to-infinity} (Proposition~\ref{prop:non-escape}).
\end{remark}

\begin{lemma}[Existence of good sets]\label{lem:good-set-existence}
Let $(M^{n+1}, g)$ be a complete Riemannian manifold with non-empty boundary and sublinear volume growth, i.e.\
\[
\liminf_{R \to \infty} \frac{\mathcal{H}^{n+1}(B_R(x) \cap M)}{R} = 0 \qquad \text{for some } x \in M.
\]
Then there exists a good set $U \subset M$ with $0 < W_g(U) < \infty$.
\end{lemma}

\begin{proof}
Fix a point $x \in M$ and a radius $r > 0$ such that the local isoperimetric constant
\[
C_I := \inf\bigl\{\mathbf{M}(\partial_I E) : E \subset B_r(x),\ \mathcal{H}^{n+1}(E) = \tfrac{1}{2}\,\mathcal{H}^{n+1}(B_r(x))\bigr\} > 0.
\]
Let $d_x: M \to [0, \infty)$ denote the distance function from $x$. Applying the slicing theorem (Theorem~\ref{thm:slicing}) to the current $T = \llbracket M \rrbracket$ and the Lipschitz function $f = d_x$ with $|\nabla d_x| \leq 1$, we obtain
\[
\int_0^\infty \mathbf{M}(\langle \llbracket M \rrbracket, d_x, R \rangle)\, dR \leq \mathcal{H}^{n+1}(M \cap B_S(x))
\]
for any $S > 0$. The sublinear growth assumption therefore allows us to pick a regular value $R > r$ such that
\[
\mathbf{M}(\langle \llbracket M \rrbracket, d_x, R \rangle) \leq \frac{C_I}{100}.
\]
Set $U := B_R(x) \cap M$. Since $\partial_I U$ is supported on $\partial B_R(x) \cap \mathrm{int}(M)$, we have $\mathbf{M}(\partial_I U) \leq \mathbf{M}(\langle \llbracket M \rrbracket, d_x, R \rangle) \leq C_I/100$. Any sweepout of $U$ must bisect volume in $B_r(x)$ at some time, so $W_\partial(U) \geq C_I$. Since $\mathbf{M}(\partial_I U) \leq \frac{1}{10} W_\partial(U)$, the set $U$ is good. Finiteness $W_g(U) < \infty$ follows from the existence of a sweepout by slices $\langle \llbracket M \rrbracket, d_x, R' \rangle$ for $R' \in [0, R]$.
\end{proof}
