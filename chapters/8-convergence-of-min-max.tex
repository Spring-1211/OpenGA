\chapter{Convergence of a min--max sequence to a minimal hypersurface}\label{ch:convergence-of-min-max}

This chapter completes the proof of the main theorem by showing that the min-max sequence constructed in the previous chapters converges to a smooth free-boundary minimal hypersurface. The argument combines three key ingredients: the pull-tight procedure adapted to the complete setting, the almost minimizing property in relative annuli, and the free-boundary regularity theory of Li--Zhou \cite{LZ17}.

Throughout this chapter we use vector fields in $\mathfrak{X}_{\mathrm{tan}}(M)$ (tangent to $\partial M$). Varifold convergence is understood in $M$, so ``stationary'' means stationary with free boundary.

Starting from a good sweepout of the good set constructed in Chapter~\ref{ch:nested-sweepout}, we perform a pull--tight deformation on an exhaustion of $M$, select slices that are almost minimizing in small relative annuli, and pass to the limit. The free boundary regularity theory of Li--Zhou~\cite{LZ17}, together with the almost minimizing property, then yields a smooth almost properly embedded free boundary minimal hypersurface.

\section{Pulled--tight sequences and min--max varifolds}\label{sec:pull-tight}
Fix a good set $U\subset M$.  Proposition~\ref{prop:nested-sweepout} produces good sweepouts with maximal area arbitrarily close to $W_g(U)$.  To prevent loss of mass at infinity we perform the pull--tight deformation on a bounded exhaustion of $M$.

Let $U_0\Subset U_1\Subset\cdots\Subset M$ be bounded relative open sets with $M=\bigcup_i U_i$ and $U\Subset U_0$.  Since each $U_i$ is precompact in $M$, the closure $\overline{U_i}$ is compact and therefore admits uniform bounds on Ricci curvature, the second fundamental form of $\partial M \cap \overline{U_i}$, and a positive lower bound on the injectivity radius, so the monotonicity formula and Li--Zhou's estimates apply on each $U_i$.  For each $i$ choose $W_i$ with $U_i\Subset W_i\Subset U_{i+1}$.  Set $C:=2W_g(U)$ and define
\[
\mathcal{V}^C(\Omega):=\bigl\{V\in\mathcal{V}_n(M):\operatorname{spt}\|V\|\subset\overline\Omega,\;\|V\|(M)\le C\bigr\},
\]
\[
\mathcal{V}^C_\infty(\Omega):=\bigl\{V\in\mathcal{V}^C(\Omega): V\text{ is stationary in }\Omega\text{ with free boundary}\bigr\}.
\]
Both sets are compact under the varifold distance function $\mathbf{F}$ (see \cite[Lemma~3.4]{LZ17}).

For each $X\in\mathfrak{X}_{\mathrm{tan}}(M)$, let $\{f_t\}_{t\in[0,1]}$ be the flow generated by $X$.  The induced map $H^X:[0,1]\times\mathcal{V}_n(M)\to\mathcal{V}_n(M)$ defined by $H^X(t,V)=(f_t)_\sharp V$ is continuous with respect to $\mathbf{F}$.  Similarly, the pushforward on relative cycles $H^X(t,\tau)=[(f_t)_\sharp T]$ where $T\in\tau$ gives a continuous map $H^X:[0,1]\times\mathcal{Z}_n(M,\partial M;\mathbb{Z}_2)\to\mathcal{Z}_n(M,\partial M;\mathbb{Z}_2)$.

\begin{lemma}[Extension of the pull--tight map]\label{lem:extend-pull-tight}
For every $i$ there exist a continuous map $\Phi_{U_i,W_i}:\mathcal{V}^C(M)\to\mathcal{V}^C(M)$ and decreasing sequences $\{\tau_k(U_i)\}$, $\{\varepsilon_k(U_i)\}$ converging to $0$ with the following properties for all $V\in\mathcal{V}^C(M)$:
\begin{enumerate}
    \item $\|\Phi_{U_i,W_i}(V)\|(M)\le \|V\|(M)$;
    \item if $\|V\|(\overline{W}_i)\le 5\,\mathbf{M}(\partial_IU)$ then $\Phi_{U_i,W_i}(V)=V$;
    \item if $\|V\|(\overline{W}_i)\ge 9\,\mathbf{M}(\partial_IU)$ and
    \[
    \mathbf{F}_{U_i}\bigl(V\llcorner Gr_n(U_i),\mathcal{V}^C_\infty(U_i)\bigr)\in\bigl[2^{-(k+1)},2^{-k}\bigr],
    \]
    then $\|\Phi_{U_i,W_i}(V)\|(M)\le \|V\|(M)-\varepsilon_k(U_i)$ and $\mathbf{F}\bigl(\Phi_{U_i,W_i}(V),V\bigr)\le \tau_k(U_i)$.
\end{enumerate}
\end{lemma}

\begin{proof}
The argument follows \cite[Proposition~4.17]{LZ17}, adapted to each bounded set $U_i$ in place of the whole manifold.  The key point is that, although $M$ is non-compact and $\mathcal{V}^C(M)$ need not be compact, each $U_i$ is bounded, so $\mathcal{V}^C(U_i)$ and $\mathcal{V}^C_\infty(U_i)$ \emph{are} compact under $\mathbf{F}$.  The tightening construction of Li--Zhou therefore applies on $U_i$: one defines a continuous map $\Psi:\mathcal{V}^C(U_i)\to\mathfrak{X}_{\mathrm{tan}}(M)$ such that $\delta V(\Psi(V))=0$ if $V\in\mathcal{V}^C_\infty(U_i)$ and $\delta V(\Psi(V))<0$ otherwise, together with a continuous function $h:\mathcal{V}^C(U_i)\to[0,1]$ satisfying $h(V)=0$ iff $V\in\mathcal{V}^C_\infty(U_i)$.

To ensure that the resulting map $\Phi_{U_i,W_i}$ only modifies varifolds inside $W_i$, we truncate the vector fields: choose a smooth cutoff $\rho_i\in C_c^\infty(W_i)$ with $\rho_i\equiv 1$ on $\overline{U}_i$ and replace each $\Psi(V)$ by $\rho_i\,\Psi(V)\in\mathfrak{X}_{\mathrm{tan}}(M)$.  The flow generated by $\rho_i\,\Psi(V)$ is the identity outside $W_i$, so $\Phi_{U_i,W_i}(V)$ agrees with $V$ on $M\setminus W_i$.  The mass-decreasing property (3) is preserved because the flow still acts as the full tightening flow on $U_i$, while property (2) ensures that $\Phi_{U_i,W_i}$ leaves invariant any varifold with small mass in $\overline{W}_i$.
\end{proof}

\begin{remark}[Pull-tight preserves the good sweepout structure]\label{rem:pt-preserves-good}
An important feature of the pull-tight construction is that it does not move mass from $U$ to $\partial_I U$. This follows from two observations:
\begin{enumerate}[label=\textup{(\roman*)}]
    \item By property (2), if a slice has mass $\leq 5\,\mathbf{M}(\partial_I U)$ in $\overline{W}_i$, then $\Phi_{U_i, W_i}$ acts as the identity on that slice. Since endpoints of a good sweepout have mass $\leq 5\,\mathbf{M}(\partial_I U)$ by Definition~\ref{def:good-sweepout}, the pull-tight procedure leaves the endpoints unchanged.
    \item The flow is generated by vector fields in $\mathfrak{X}_{\mathrm{tan}}(M)$ supported in $W_i$. Since $U \Subset U_0 \Subset W_0$, the cutoff $\rho_0$ vanishes near $\partial_I U$, so the flow does not transport mass across $\partial_I U$.
\end{enumerate}
Therefore the pulled-tight sequence $\{\mathbf{P}_i(\Gamma_t^i)\}$ remains a sequence of good sweepouts, and the non-escape property (Proposition~\ref{prop:non-escape}) applies to the limit.
\end{remark}

Let $\{\Gamma_t^i\}$ be a sequence of good sweepouts with $\sup_t \mathbf{M}(\Gamma_t^i)\downarrow W_g(U)$.  Set
\[
\mathbf{P}_i:=\Phi_{U_0,W_0}\circ\Phi_{U_1,W_1}\circ\cdots\circ\Phi_{U_i,W_i}.
\]

\begin{proposition}[Pulled--tight minimizing sequence]\label{prop:pt-minimizing}
After passing to a subsequence, $\{\mathbf{P}_i(\Gamma_t^i)\}$ is a minimizing sequence of good sweepouts with
\[
\lim_{i\to\infty}\ \sup_{t\in[0,1]}\mathbf{M}\bigl(\mathbf{P}_i(\Gamma_t^i)\bigr)=W_g(U),
\]
and any min--max sequence $\mathbf{P}_i(\Gamma_{t_i}^i)$ admits a subsequence converging to a varifold in $\mathcal{V}^C_\infty(M)$, i.e.\ stationary in $M$ with free boundary.
\end{proposition}

\begin{proof}
Property (3) in Lemma~\ref{lem:extend-pull-tight} guarantees that if the restriction of a slice to some $U_j$ stays a definite $\mathbf{F}_{U_j}$--distance from $\mathcal{V}^C_\infty(U_j)$, then the composed map decreases its mass by at least $\varepsilon_k(U_j)$.

We now describe the diagonal argument. For each fixed $j$, the sequence $\{\mathbf{P}_i(\Gamma_{t_i}^i)\}_{i \geq j}$ has uniformly bounded mass (by $C = 2W_g(U)$) and is supported in $\overline{U}_j$, so by compactness of $\mathcal{V}^C(U_j)$ under $\mathbf{F}$, we can extract a subsequence converging in $\mathbf{F}_{U_j}$ to some $V_j \in \mathcal{V}^C(U_j)$. Moreover, property (3) ensures that $V_j \in \mathcal{V}^C_\infty(U_j)$, i.e., $V_j$ is stationary in $U_j$ with free boundary.

Applying a standard diagonal argument: first extract a subsequence $\{i^{(1)}_k\}$ converging in $\mathbf{F}_{U_1}$, then a further subsequence $\{i^{(2)}_k\}$ converging in $\mathbf{F}_{U_2}$, and so on. The diagonal sequence $\{i^{(k)}_k\}$ converges in $\mathbf{F}_{U_j}$ for every $j$. Since $M = \bigcup_j U_j$ and the exhaustion is nested, the limit varifold $V$ is well-defined on all of $M$ and satisfies $\delta V(X) = 0$ for every $X \in \mathfrak{X}_{\mathrm{tan}}(M)$ with $\mathrm{spt}(X) \subset U_j$ for some $j$. Since every compactly supported tangential vector field is supported in some $U_j$, we conclude that $V$ is stationary in $M$ with free boundary.
\end{proof}

\section{Almost minimizing varifolds with free boundary}\label{sec:am-varifolds}

We recall the notion of almost minimizing varifolds with free boundary, following Li--Zhou~\cite[Section~4]{LZ17}.

\begin{definition}[Almost minimizing with free boundary {\cite[Definition~4.18]{LZ17}}]\label{def:am-free-bdy}
Let $U\subset M$ be relatively open.  For $\varepsilon,\delta>0$, define $\mathscr{A}_n(U;\varepsilon,\delta;\mathbf{F})$ to be the set of all $\tau\in\mathcal{Z}_n(M,\partial M;\mathbb{Z}_2)$ such that whenever $\tau=\tau_0,\tau_1,\ldots,\tau_m\in\mathcal{Z}_n(M,\partial M;\mathbb{Z}_2)$ is a discrete sequence with
\begin{itemize}
    \item $\operatorname{spt}(\tau_i-\tau)\subset U$,
    \item $\mathbf{F}(\tau_{i+1}-\tau_i)\le\delta$,
    \item $\mathbf{M}(\tau_i)\le\mathbf{M}(\tau)+\delta$ for $i=1,\ldots,m$,
\end{itemize}
then $\mathbf{M}(\tau_m)\ge\mathbf{M}(\tau)-\varepsilon$.

A varifold $V\in\mathcal{V}_n(M)$ is \emph{almost minimizing in $U$ with free boundary} if there exist sequences $\varepsilon_i\to 0$, $\delta_i\to 0$, and $\tau_i\in\mathscr{A}_n(U;\varepsilon_i,\delta_i;\mathbf{F})$ with canonical representatives $T_i\in\tau_i$ such that $\mathbf{F}(|T_i|,V)\le\varepsilon_i$.
\end{definition}

\begin{definition}[Almost minimizing in small annuli {\cite[Definition~5.1]{LZ17}}]\label{def:am-small-annuli}
For $p\in M$ and $0<s<r$, set
\[
\mathcal{A}_{s,r}(p):=
\begin{cases}
B_r(p)\setminus\overline{B}_s(p) & \text{if } p\in\operatorname{int}(M),\\
B_r^+(p)\setminus\overline{B}^+_s(p) & \text{if } p\in\partial M,
\end{cases}
\]
where $B_r^+(p):=B_r(p)\cap M$.  A varifold $V\in\mathcal{V}_n(M)$ is \emph{almost minimizing in small annuli with free boundary} if for each $p\in M$ there exists $r_{\mathrm{am}}(p)>0$ such that $V$ is almost minimizing in $\mathcal{A}_{s,r}(p)$ with free boundary for all $0<s<r\le r_{\mathrm{am}}(p)$.
\end{definition}

\begin{proposition}[Stationarity {\cite[Proposition~4.21]{LZ17}}]\label{prop:am-stationary}
If $V\in\mathcal{V}_n(M)$ is almost minimizing in a relatively open set $U\subset M$ with free boundary, then $V$ is stationary in $U$ with free boundary.
\end{proposition}

The main regularity result is the following:

\begin{theorem}[Regularity of almost minimizing varifolds {\cite[Theorem~6.2]{LZ17}}]\label{thm:main-regularity}
Let $2\le n\le 6$.  Suppose $V\in\mathcal{V}_n(M)$ is stationary in $M$ with free boundary and almost minimizing in small annuli with free boundary.  Then there exist $N\in\mathbb{N}$, positive integers $m_1,\ldots,m_N$, and smooth, connected, almost properly embedded free boundary minimal hypersurfaces $(\Sigma_j,\partial\Sigma_j)\subset(M,\partial M)$ such that
\[
V=\sum_{j=1}^N m_j\,|\Sigma_j|.
\]
\end{theorem}

\begin{remark}[Dimension restriction]\label{rem:dimension-restriction}
The restriction $2 \leq n \leq 6$ arises from the regularity theory. For $n \leq 6$, the singular set of an almost minimizing varifold with free boundary is empty by \cite[Theorem~6.2]{LZ17}, which relies on Wickramasekera's regularity theory~\cite{Wic14}. For $n = 7$, the singular set is discrete, and for $n \geq 8$, the singular set has Hausdorff dimension at most $n - 7$. In those cases, the main theorem still holds with the conclusion that $\Sigma$ is smooth and embedded \emph{away from a singular set of codimension at least $7$}.
\end{remark}

\section{Limit varifolds and proof of the main theorem}\label{sec:main-proof}

\subsection{Almost minimizing selection in the non-compact setting}

Let $\{\Gamma_i\}$ be the pulled-tight min--max sequence produced in Section~\ref{sec:pull-tight}.  By Proposition~\ref{prop:pt-minimizing}, we may extract a subsequence (not relabelled) such that the induced varifolds converge to some $V\in\mathcal{V}^C_\infty(M)$.

In the compact setting, Li--Zhou~\cite[Theorem~5.3]{LZ17} obtain the almost minimizing property from the combinatorial selection argument of Almgren--Pitts~\cite{Pit81}.  In our non-compact setting, the key observation is that the combinatorial argument is \emph{local}: for each $p\in M$, one only needs to deform sweepouts inside a bounded neighbourhood of $p$.  Since each element of the exhaustion $U_j$ is bounded, the pull--tight map $\mathbf{P}_i$ restricts to a finite composition of maps $\Phi_{U_k,W_k}$ on any fixed bounded region, and the combinatorial argument of \cite[Theorem~5.3]{LZ17} applies on each $U_j$ without modification.  A diagonal argument over $j$ then yields:

\begin{proposition}[Almost minimizing selection]\label{prop:am-selection}
After passing to a further subsequence, $V$ is almost minimizing in small annuli with free boundary in the sense of Definition~\ref{def:am-small-annuli}.
\end{proposition}

\subsection{Regularity of the limit varifold}

\begin{proposition}\label{prop:regular-limit}
There exist finitely many connected, smooth, almost properly embedded free boundary minimal hypersurfaces $(\Sigma_j,\partial\Sigma_j)\subset (M,\partial M)$ and positive integers $m_j$ such that
\[
V=\sum_{j=1}^N m_j|\Sigma_j|.
\]
\end{proposition}

\begin{proof}
By Proposition~\ref{prop:am-selection}, $V$ is almost minimizing in small annuli with free boundary, hence stationary in $M$ with free boundary by Proposition~\ref{prop:am-stationary}.  The regularity theory of Li--Zhou (Theorem~\ref{thm:main-regularity}) then applies.  We verify that the non-compactness of $M$ causes no difficulty: the proof of Theorem~\ref{thm:main-regularity} in \cite[Section~6]{LZ17} consists of three parts---the good replacement property, the classification of tangent cones, and the boundary regularity---all of which are \emph{local} arguments carried out in a fixed ball $B_r(p)\cap M$.  In particular:
\begin{enumerate}
    \item The replacement construction (\cite[Proposition~6.3]{LZ17}) only modifies $V$ inside a compact annular region $\overline{\mathcal{A}}_{s,r}(p)\cap M$, where area minimization in a bounded domain is well-posed.
    \item The tangent cone analysis (\cite[Proposition~6.11]{LZ17}) uses the monotonicity formula for stationary varifolds with free boundary (\cite[Theorem~2.3]{LZ17}) and Allard-type regularity, both of which are local.
    \item The boundary regularity (\cite[Theorem~6.2]{LZ17}) proceeds by blow-up at a point $p\in\operatorname{spt}\|V\|\cap\partial M$, which only sees the local geometry near $p$.
\end{enumerate}
Therefore the conclusion of Theorem~\ref{thm:main-regularity} holds for $V$ in the complete, non-compact manifold $M$.
\end{proof}

\subsection{The main theorem}

We can now state and prove the main result of this thesis.

\begin{theorem}[Main theorem]\label{thm:main-complete}
Let $(M^{n+1},g)$ be a complete Riemannian manifold with non-empty smooth boundary $\partial M$, $2\le n\le 6$.  Let $U\subset M$ be a \emph{good set} (Definition~\ref{def:good-set}), i.e., a bounded open set of finite perimeter satisfying
\[
\mathbf{M}(\partial_I U) \leq \frac{1}{10}\, W_\partial(U),
\]
with $\overline{U}\cap\partial M\neq\varnothing$.  Then there exists a smooth, almost properly embedded free boundary minimal hypersurface $(\Sigma,\partial\Sigma)\subset(M,\partial M)$ with $\partial\Sigma\subset\partial M$, $\Sigma$ meeting $\partial M$ orthogonally, and
\[
\mathcal{H}^n(\Sigma)\le W_\partial(U)+\mathbf{M}(\partial_IU).
\]
Moreover, $\Sigma\cap\overline{U}\neq\varnothing$.
\end{theorem}

\begin{remark}\label{rem:sigma-extends}
The hypersurface $\Sigma$ may extend beyond $\overline{U}$, but the good set condition ensures $\Sigma \cap \overline{U} \neq \varnothing$ with definite area. More precisely, by Proposition~\ref{prop:non-escape}, we have $\mathcal{H}^n(\Sigma \cap \overline{U}) \geq \varepsilon(U) > 0$ for an explicit constant $\varepsilon(U)$ depending on $U$.
\end{remark}

\begin{proof}
The proof combines all preceding chapters.

\medskip
\noindent\textbf{Step 1: Construction of the min--max sequence.}
Since $U$ is a good set, we have $W_g(U) > 0$. Indeed, the good set condition $\mathbf{M}(\partial_I U) \leq \frac{1}{10} W_\partial(U)$ implies $W_\partial(U) \geq 10\,\mathbf{M}(\partial_I U) > 0$ (since $U$ is non-empty and has non-trivial interior boundary). By Proposition~\ref{prop:width-inequalities}, $W_g(U) \geq W_\partial(U) - \mathbf{M}(\partial_I U) \geq \frac{9}{10} W_\partial(U) > 0$.

By Proposition~\ref{prop:nested-sweepout}, there exists a sequence of good sweepouts $\{\Gamma_t^i\}$ of $U$ with $\sup_t\mathbf{M}(\Gamma_t^i)\downarrow W_g(U)$.  Applying the pull--tight procedure of Section~\ref{sec:pull-tight}, we obtain a minimizing sequence $\{\mathbf{P}_i(\Gamma_t^i)\}$ whose min--max varifold limit $V$ is stationary in $M$ with free boundary (Proposition~\ref{prop:pt-minimizing}).

\medskip
\noindent\textbf{Step 2: Almost minimizing and regularity.}
By Proposition~\ref{prop:am-selection}, $V$ is almost minimizing in small annuli with free boundary.  Proposition~\ref{prop:regular-limit} then gives
\[
V=\sum_{j=1}^N m_j\,|\Sigma_j|,
\]
where each $(\Sigma_j,\partial\Sigma_j)\subset(M,\partial M)$ is a smooth, connected, almost properly embedded free boundary minimal hypersurface meeting $\partial M$ orthogonally.

\medskip
\noindent\textbf{Step 3: Area bound and non-escape.}
Set $\Sigma:=\bigcup_{j=1}^N\Sigma_j$.  The area bound follows from the min--max construction:
\[
\mathcal{H}^n(\Sigma)\le\|V\|(M)=W_g(U)\le W_\partial(U)+\mathbf{M}(\partial_IU).
\]
The non-escape estimate (Proposition~\ref{prop:non-escape}) guarantees that for every $\delta>0$,
\[
\mathcal{H}^n(\Sigma\cap N_\delta(U))\ge\frac{\varepsilon(U)}{2}>0,
\]
so in particular $\Sigma\cap\overline{U}\neq\varnothing$.

\medskip
\noindent\textbf{Step 4: Non-triviality of the free boundary.}
It remains to show $\partial\Sigma\neq\varnothing$.  The good sweepouts of $U$ are nested sweepouts in the sense of Definition~\ref{def:nested-sweepout}: each slice $\Gamma_t^i=\partial_I\Omega_t^i$ is the interior boundary of an open set $\Omega_t^i$ with $\Omega_0^i=\varnothing$ and $\Omega_1^i=U$.  Since $\overline{U}\cap\partial M\neq\varnothing$, the continuous family $\{\Omega_t^i\}$ sweeps through the boundary region, and the relative homology class $[\partial_I U]\in H_n(M,\partial M;\mathbb{Z}_2)$ is non-trivial (as $U$ has non-empty intersection with $\partial M$).

Suppose for contradiction that $\partial\Sigma=\varnothing$.  Then each $\Sigma_j$ is a closed minimal hypersurface in $\operatorname{int}(M)$, and $V$ is supported away from $\partial M$.  We derive a contradiction as follows.  Since $\overline{U}\cap\partial M\neq\varnothing$ and the sweepout $\{\Omega_t^i\}$ is continuous in the $L^1$ topology with $\Omega_0^i=\varnothing$ and $\Omega_1^i=U$, by the intermediate value theorem there exists $t^*_i$ with $\mathcal{H}^{n+1}(\Omega_{t^*_i}^i\cap U)\in\bigl(\varepsilon_0,\,\mathcal{H}^{n+1}(U)-\varepsilon_0\bigr)$.  The isoperimetric profile of $\overline{U}$ (cf.\ Proposition~\ref{prop:non-escape}) then gives $\mathcal{H}^n(\partial_I\Omega_{t^*_i}^i\cap U)\geq 2\varepsilon(U)>0$.  In particular, the slice $\Gamma_{t^*_i}^i=\partial_I\Omega_{t^*_i}^i$ has definite mass near $\partial M\cap\overline{U}$.  Passing to the varifold limit, $\|V\|(\overline{U}\cap\partial M)>0$ by lower semicontinuity, contradicting $\operatorname{spt}\|V\|\cap\partial M=\varnothing$.  (This argument is the free-boundary analogue of \cite[Lemma~5.4]{LZ17}, adapted to the non-compact setting with $\mathbb{Z}_2$ coefficients; the key input is the non-triviality of the mod~$2$ relative homology class $[\partial_I U]\in H_n(M,\partial M;\mathbb{Z}_2)$.)

Therefore $\partial\Sigma\neq\varnothing$, and at least one component $\Sigma_j$ is a free boundary minimal hypersurface with $\partial\Sigma_j\subset\partial M$.
\end{proof}

As an immediate corollary, we obtain existence under a volume growth condition.

\begin{corollary}\label{cor:sublinear-existence}
Let $(M^{n+1},g)$ be a complete Riemannian manifold with non-empty smooth boundary $\partial M$, $2\le n\le 6$, and sublinear volume growth:
\[
\liminf_{R\to\infty}\frac{\mathcal{H}^{n+1}(B_R(x)\cap M)}{R}=0\qquad\text{for some }x\in M.
\]
Suppose in addition that $\partial M$ is non-compact, or more generally that there exists a good set $U$ with $\overline{U}\cap\partial M\neq\varnothing$.  Then $M$ contains a smooth, almost properly embedded free boundary minimal hypersurface.
\end{corollary}

\begin{proof}
By Lemma~\ref{lem:good-set-existence}, the sublinear growth condition guarantees the existence of a good set $U\subset M$.  If $\overline{U}\cap\partial M\neq\varnothing$, the result follows from Theorem~\ref{thm:main-complete}.
\end{proof}
